- Update file: 
	- git add folder name or file name
	- git commit -m "a text to descripe changes"
	- git push -u origin master
- Delete file:
	- rm "filename"
	- git add "filename"
	- git commit -m "delete filename"
	- git push -u origin master
- Change file name
	- git mv -f oldfilename newfilename
